#include <string>
#include <vector>
#include <iostream>
#include <algorithm>

using namespace std;

int solution(vector<vector<int>> dots) {
    int answer = 0;
    
    int x1, x2, y1, y2;
    
    sort(dots.begin(), dots.end());
    
    x1 = dots[0][0];
    y1 = dots[0][1];
    
    x2 = dots[3][0];
    y2 = dots[3][1];
    
    int x = x2 - x1;
    int y = y2 - y1;
    answer = x * y;
    
    return answer;
}
